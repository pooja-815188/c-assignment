#include<stdio.h>
void insertion(int arr[],int n);
void print(int arr[],int n);
int main()
{
    int n=5;
    int arr[]={5,3,4,1,2};
    int i,j,temp;
    insertion(arr,n);
    print(arr,n);
}
void insertion(int arr[],int n)
{  
    int i,j,temp;
    for(i=1;i<n;i++)
    {
        temp=arr[i];
        j=i-1;
        while(j>=0 && arr[j]>temp)
        {
            arr[j+1]=arr[j];
            j--;
        }
        arr[j+1]=temp;
    }
}
void print(int arr[],int n)
{
    int i;
    for(i=0;i<n;i++)
    {
        printf("%d\n",arr[i]);
    }
}
