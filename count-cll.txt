#include<stdio.h>
#include<stdlib.h>

struct sample 
{
    int data;
    struct sample *next;
};


void add_at_beg(struct sample**h,int num)
{
struct sample*temp=(struct sample*)malloc(sizeof(struct sample));
if(*h==NULL)
    {
    temp->data=num;
    temp->next=temp;
    *h=temp;
    }
else
{
    temp->data=num;
    temp->next=(*h)->next;
    (*h)->next=temp;
}
}
