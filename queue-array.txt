#include<stdio.h>
int queue[100],i,n,choice,front=-1,rear=-1,x;

void enqueue()
{
    if(rear==n-1)
    {
        printf("queue is overflow\n");
    }
    else
    {
        printf("enter a elemts in queue\n");
        scanf("%d",&x); 
        if(front==-1 && rear==-1)
        {
            rear++;
            front++;
            queue[rear]=x;
        }
        else
        {
            rear++;
            queue[rear]=x;
        }
    }
}
void dequeue()
{
    if(front > rear)
    {
        printf("queue is under flow\n");
    }
    else
        {
             printf("deleted elemts is=%d\n",queue[front]);
             front=front+1;
         }
}
void display()
{
    if(front==-1)
    {
        printf("queue is empty\n");
    }
    else
    {
        printf("elemets in queue\n");
        for(i=front;i<=rear;i++)
        printf("%d\n",queue[i]);
    } 
}
int main()
{
   printf("enter a size of queue\n");
   scanf("%d",&n);
   printf("***queue operations are***\n");
   printf("1.enqueue\t\t2.dequeue\n3.display\t\t4.exit\n");

   while(choice!=4)
   {
       printf("\nenter a choice\n");
       scanf("%d",&choice);

       switch(choice)
       {
           case 1:
           {
               enqueue();
               break;
           }
           case 2:
           {
               dequeue();
               break;
           }
           case 3:
           {
               display();
               break;
           }
           case 4:
           {
               printf("exiting..\n");
               break;
           }
           default:
           {
               printf("enter a valid choice\n");
           }
       }
   }
}
